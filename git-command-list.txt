6. git branch
	1. git branch <branch-name> - creates a new branch
	2. git branch  --lists all the branches the * prefixed the branch-name denotes the current branch  which you have checked out.
	3. git branch -d <branch-name> -- deletes the branche
	4. git branch -D <branch-name> --force deletes the unmerged branch
	5. git branch –merged –shows all the branches that are merged
	6. git branch –no-merged –shows all the branch that is/are unmerged
	7. git branch -v –shows the last commit on each branch
